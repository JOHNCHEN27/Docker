#路由转发配置
worker_processes  1;

events {
    worker_connections  1024;
}

http {
    include       mime.types;
    default_type  application/json;

    sendfile        on;
    
    keepalive_timeout  65;

    server {
        listen      80;
        # 服务器名称 当访问ks.lncanswer.cn域名时触发
        server_name   ks.lncanswer.cn;

        location / {
        proxy_set_header Host $host;
        proxy_pass    http://nginx:8081; # 当你访问80端口可以实现向8081端口转发

  }
    }
}


#路由转发配置
worker_processes  1;

events {
    worker_connections  1024;
}

http {
    include       mime.types;
    default_type  application/json;

    sendfile        on;
    
    keepalive_timeout  65;

    server {
        listen       8081;
        # 指定前端项目所在的位置
        location / {
            root /usr/share/nginx/html/dist;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
        location /api {
            rewrite /api/(.*)  /$1 break;
            proxy_pass http://secondhandgoods:8080;
        }
    }
}
